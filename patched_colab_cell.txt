def resize_for_memory(image, max_size=(512, 768)):
    """Resize image to fit in memory (same as your original function)"""
    image.thumbnail(max_size, Image.Resampling.LANCZOS)
    return image

def base64_to_image(base64_string):
    """Convert base64 string to PIL Image"""
    # Remove data URL prefix if present (e.g., "data:image/jpeg;base64,")
    if ',' in base64_string:
        base64_string = base64_string.split(',')[1]

    image_bytes = base64.b64decode(base64_string)
    image = Image.open(io.BytesIO(image_bytes)).convert("RGB")
    return image

def image_to_base64(image):
    """Convert PIL Image to base64 string for returning"""
    buffered = io.BytesIO()
    image.save(buffered, format="PNG")
    return base64.b64encode(buffered.getvalue()).decode()

jobs = {}
job_queue = queue.Queue()

def worker():
    """Background worker to process try-on jobs sequentially"""
    while True:
        try:
            job_id = job_queue.get()
            if job_id is None:
                break
                
            job = jobs[job_id]
            job['status'] = 'processing'
            
            print(f"\n" + "="*50)
            print(f"🔄 [{time.strftime('%H:%M:%S')}] Processing job {job_id}")
            
            start_time = time.time()
            
            # Extract images
            custom_person = job['person_image']
            custom_garment = job['garment_image']
            category = job['category']
            
            # Apply resizing
            custom_person = resize_for_memory(custom_person)
            custom_garment = resize_for_memory(custom_garment)
            
            print(f"📏 Resized images: Person {custom_person.size}, Garment {custom_garment.size}")
            
            # Clear cache
            torch.cuda.empty_cache()
            gc.collect()
            
            # Run inference
            print(f"🔄 Running try-on with your images...")
            custom_result = pipeline(
                person_image=custom_person,
                garment_image=custom_garment,
                category=category
            )
            
            # Extract the result image
            if hasattr(custom_result, 'images'):
                result_image = custom_result.images[0]
            elif isinstance(custom_result, dict) and 'images' in custom_result:
                result_image = custom_result['images'][0]
            elif isinstance(custom_result, list):
                result_image = custom_result[0]
            else:
                result_image = custom_result
                
            # Calculate processing time
            processing_time = time.time() - start_time
            
            # Save the result
            timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
            filename = f"tryon_result_{job_id[:8]}_{timestamp}.png"
            result_image.save(filename)
            print(f"💾 Saved result to {filename}")
            
            # Automatically display the image in Colab cell output (if auto_display exists)
            try:
                auto_display_result(result_image, custom_person, custom_garment, processing_time)
            except Exception as e:
                print(f"⚠️ Auto display failed (non-fatal): {str(e)}")
            
            # Convert result to base64
            result_base64 = image_to_base64(result_image)
            
            # Update job status
            job['status'] = 'completed'
            job['result_image'] = f"data:image/png;base64,{result_base64}"
            job['result_filename'] = filename
            job['processing_time'] = round(processing_time, 2)
            
            print(f"✅ Job {job_id} complete in {processing_time:.2f} seconds")
            print("="*50)
            
            # Free up memory
            del job['person_image']
            del job['garment_image']
            
        except Exception as e:
            print(f"❌ Error processing job {job_id}: {str(e)}")
            import traceback
            traceback.print_exc()
            if 'job_id' in locals() and job_id in jobs:
                jobs[job_id]['status'] = 'failed'
                jobs[job_id]['error'] = str(e)
        finally:
            if 'job_id' in locals():
                job_queue.task_done()

# Start worker thread

# ===== API ENDPOINT THAT QUEUES YOUR TRY-ON JOB =====
@app.route('/tryon', methods=['POST'])
def tryon():
    """
    This endpoint parses the request, queues the job, and returns a job_id.
    """
    try:
        data = request.json

        if not data:
            return jsonify({'error': 'No data provided'}), 400

        if 'job_id' in data:
            # Check status of an existing job
            job_id = data['job_id']
            if job_id not in jobs:
                return jsonify({'success': False, 'error': 'Job not found'}), 404
            
            job = jobs[job_id]
            status = job['status']
            
            response = {
                'success': True,
                'job_id': job_id,
                'status': status
            }
            
            if status == 'completed':
                response['result_image'] = job['result_image']
                response['result_filename'] = job['result_filename']
                response['processing_time'] = job['processing_time']
                response['message'] = "Processed successfully"
                response['category'] = job.get('category', 'tops')
            elif status == 'failed':
                response['error'] = job.get('error', 'Unknown error')
                
            return jsonify(response)

        if 'person_image' not in data or 'garment_image' not in data:
            return jsonify({'error': 'Both person_image and garment_image required'}), 400

        print(f"\n📥 [{time.strftime('%H:%M:%S')}] Received API request - Queueing Job")

        # Convert base64 to images immediately so we validate them early
        custom_person = base64_to_image(data['person_image'])
        custom_garment = base64_to_image(data['garment_image'])
        category = data.get('category', 'tops')

        job_id = str(uuid.uuid4())
        
        jobs[job_id] = {
            'id': job_id,
            'status': 'pending',
            'person_image': custom_person,
            'garment_image': custom_garment,
            'category': category,
            'created_at': time.time()
        }
        
        job_queue.put(job_id)
        
        return jsonify({
            'success': True,
            'message': 'Job queued successfully',
            'job_id': job_id,
            'status': 'pending'
        })

    except Exception as e:
        print(f"❌ Error queueing job: {str(e)}")
        import traceback
        traceback.print_exc()
        return jsonify({
            'success': False,
            'error': str(e)
        }), 500
